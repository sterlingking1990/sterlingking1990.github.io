import pandas as pd
import numpy as np
from pandas import Series

movie_df=pd.read_csv("fandango_score_comparison.csv")
movie_dt=movie_df.dtypes #this stores the tuple (column_name,type) into the variable movie_dt
#analyse the type and make sure to output only ones with float64
float_columns=movie_dt[movie_dt.values=='float64'].index #this returns the column_names with float 64 types and places in a list
#we are to access only the Fandango_stars and fandango_Ratingvalue from all columns
movie_float_columns=movie_df[float_columns] #this selects all the columns with float 64
fandango_sr=movie_float_columns[['Fandango_Stars','Fandango_Ratingvalue']]
#find average of the fandango_sr and store in a column named fandango_avg_rating
fandango_avg=fandango_sr.apply(lambda x: np.sum(x)/2, axis=1)
movie_df["Fandango_avg_rating"]=fandango_avg

#write a function that will return for every movie, its fandango_avg rating
def return_fandango_avg(movie_name):
    fandango_rating_for_movie_name=movie_df[movie_df['FILM']==movie_name]
    return fandango_rating_for_movie_name["Fandango_avg_rating"]

fandango_avg=return_fandango_avg("Woman in Gold (2015)")
print("Fandango average rating based on function is " + str(fandango_avg))

#or use series to select the film and fandango_avg_rating
film_select=movie_df['FILM'].values
fandango_avg_rate=movie_df['Fandango_avg_rating'].values
film_fandango_avg_rating_series=Series(fandango_avg_rate,film_select)
film_fandango_series=film_fandango_avg_rating_series.sort_index()
print("Fandango Average rating based on series is " + str(film_fandango_series['Woman in Gold (2015)']))

#there are two ways of writing the dataframe function
#1. you can write a function that you will pass in a parameter during run time, this is a manual tool function; this function 
#will serve as a manual tool where you decide at run time how it will run, what input to use to fetch the return value 
#i.e return_fandango_avg(movie_name) is an example of a manual tool function

#2. you can write a function as a bot function, this is meant to automatically execute within the dataframe and return a result
#in the definition of this function, the input is already set as a constant and the result is will always be the same for that input
#it is diff from a manual tool function bcos the mt function will allow the user to always change input during execution
#the bot function wiil have an already set input which will execute as soon as it is applied in the dataframe.

#3. the difference also is that the bot function is applied to the dataframe, while the manual tool function has an input parameter
#passed to it

#4 use the manual tool function when you are to execute for different choices
#use the bot function when you are to automate something that does not require input i.e to normalize series or items in a dataframe,
#to create labels based on items on a particular series i.e age values; if age > 18 label = maturity etc. then these labels
#form a new series column for labelling the age values and further decision making

import pandas as pd
import numpy as np
from numpy import arange
import matplotlib.pyplot as plt

movie_dataframe=pd.read_csv("fandango_score_comparison.csv")
norm_review=pd.DataFrame()

#normalize the ratings for the various movie rating categories and store the value in another dataframe called norm_reviews
cols_to_norm=['RT_user_norm','Metacritic_user_nom','IMDB_norm','Fandango_Ratingvalue','Fandango_Stars']
for i in cols_to_norm:
    max_cols=movie_dataframe[i].max()
    norm_review[i]=movie_dataframe[i]/max_cols
#print(norm_review.iloc[0].values)    
bar_positions=arange(5)+0.75
bar_heights=norm_review[cols_to_norm].iloc[0].values
fig,ax=plt.subplots()
ax.bar(bar_position,bar_heights,0.5)
tick_positions=range(1,6)
ax.set_xticks(tick_positions)
ax.set_xticklabels(cols_to_norm,rotation=90)
ax.set_xlabel("Movie Rating Source")
ax.set_ylabel("Rating Score")
ax.set_title("Movie Rating Trends (AVENGERS: AGE OF ULTRON (2015)")
#%matplotlib inline
plt.show()
